% SLIDE 4: Three methods at a glance
% ============================================================
\begin{frame}{Three regime-detection methods}
  \vspace{14pt}
  \begin{center}
  \begin{tabular}{p{0.20\textwidth} p{0.32\textwidth} p{0.36\textwidth}}
    \toprule
    \textbf{Method} & \textbf{Family} & \textbf{What it learns from} \\
    \midrule
    \emphp{A. Markov} & Econometric state-space (Hamilton 1989) & Each factor's own return history \\[8pt]
    \emphp{B. Ridge}  & Linear ML, regularized regression & Factor lags + rolling volatilities \\[8pt]
    \emphp{C. GBM}    & Tree-based machine learning & Same features, nonlinear interactions \\
    \bottomrule
  \end{tabular}
  \end{center}
  \vspace{18pt}
  \begin{itemize}
    \item All three fit \textbf{independently per factor}, refit \textbf{annually} on expanding windows
    \item Each outputs $P(\text{favorable})$ per factor per month
    \item Same OOS discipline, same features, same backtest harness
  \end{itemize}
\end{frame}

% ============================================================
% SLIDE 5: Method A explained
% ============================================================
\begin{frame}{Method A: Markov regime switching}
  \vspace{8pt}
  Two latent states; the model infers a smoothed posterior probability of being in the favorable regime each month.
  \vspace{14pt}
  \begin{center}
  \fbox{\parbox{0.78\textwidth}{\centering
    Observation: $r^{f}_t \;\sim\; \mathcal{N}(\mu_{s_t},\, \sigma^2_{s_t})$\\[6pt]
    State transitions: $P(s_{t+1} \mid s_t)$ via $2\times 2$ matrix\\[6pt]
    Forward filter: $P(s_t \mid r_{1..t})$
  }}
  \end{center}
  \vspace{14pt}
  \begin{itemize}
    \item One model per FF5 factor, fit by maximum likelihood
    \item Favorable state = the one with higher in-sample mean
    \item Online forward filter, no future leakage
  \end{itemize}
\end{frame}

% ============================================================
% SLIDE 7: Method B explained
% ============================================================
\begin{frame}{Method B: Ridge predictive regression}
  \vspace{8pt}
  Predict next-month factor return from lagged factor returns and rolling volatilities, then map to favorable probability via sigmoid.
  \vspace{14pt}
  \begin{center}
  \fbox{\parbox{0.84\textwidth}{\centering
    $\hat{r}^{f}_{t+1} \;=\; \boldsymbol\beta^\top \mathbf{x}_t \;+\; \alpha\,\|\boldsymbol\beta\|^2$\\[6pt]
    $P(\text{favorable}_t) \;=\; \sigma\!\left( \dfrac{\hat{r}^{f}_{t+1}}{\hat{\sigma}_{\text{train}}} \right)$
  }}
  \end{center}
  \vspace{14pt}
  \begin{itemize}
    \item L2 penalty ($\alpha$) chosen by 5-fold cross-validation
    \item Features standardized within each training window
    \item One model per factor, refit annually
  \end{itemize}
\end{frame}

% ============================================================
% SLIDE 9: Method C explained
% ============================================================
\begin{frame}{Method C: gradient-boosted classifier}
  \vspace{8pt}
  Reframe as classification: label months as favorable when next-month return exceeds the in-sample median; learn nonlinear decision boundaries.
  \vspace{14pt}
  \begin{center}
  \fbox{\parbox{0.84\textwidth}{\centering
    Label: $y^{f}_t = \mathbb{1}\{r^{f}_{t+1} > \text{median}(r^{f}_{\text{train}})\}$\\[6pt]
    200 trees, depth 3, learning rate 0.05, subsample 0.8\\[6pt]
    Output: $\hat{P}(y^{f}_t = 1 \mid \mathbf{x}_t)$
  }}
  \end{center}
  \vspace{14pt}
  \begin{itemize}
    \item Hyperparameters fixed a priori (no per-window CV)
    \item Captures nonlinear interactions Ridge cannot
    \item One classifier per factor, refit annually
  \end{itemize}
\end{frame}

  \end{center}
  \vspace{18pt}
  \begin{itemize}
    \item All three see the same factor return histories, react similarly
    \item Naive averaging cannot exploit decorrelation that does not exist
    \item \emphp{The pivot:} use the timed signal as a defensive overlay, not a replacement
  \end{itemize}

% SLIDE 12: Inertia construction
% ============================================================
\begin{frame}{The \fundname{} strategy}
  \vspace{14pt}
  \begin{center}
    \large
    \begin{equation*}
      r_{\fundname{},t} \;=\; 0.5 \cdot r_{t}^{\text{static FF5}} \;+\; 0.5 \cdot r_{t}^{\text{Method C timed}}
    \end{equation*}
  \end{center}
  \vspace{16pt}
  \begin{itemize}
    \item Half of capital in static equal-weight FF5
    \item Half in Method C's GBM-driven long-short factor sleeve
    \item Method C is the timed component because it has the lowest correlation to static, providing the most diversification at the portfolio level
    \item \emphp{Robust}: equal-thirds and 0.4/0.3/0.3 weightings also produce Sharpe 0.76 to 0.80
  \end{itemize}
\end{frame}

% SLIDE 18: Why it works
% ============================================================
\begin{frame}{Why the defensive hybrid works}
  \vspace{16pt}
  \begin{itemize}
    \item \emphp{Static FF5} captures the long-run factor risk premia. Robust, well-documented, hard to beat directly.
    \item \emphp{Method C} provides regime-conditional dampening. When $P(\text{favorable})$ drops below 0.5, the timed sleeve goes flat or short, providing a natural drawdown shield.
    \item \emphp{The 50/50 mix} balances the two:
      \begin{itemize}
        \item Most of the static premium is preserved
        \item Half of capital actively manages drawdown risk
        \item Net effect: similar mean return, much lower volatility, smaller drawdowns
      \end{itemize}
    \item \emphb{Class concept}: dynamic mean-variance allocation, applied conditionally to factor portfolios
  \end{itemize}
\end{frame}

  \end{itemize}
