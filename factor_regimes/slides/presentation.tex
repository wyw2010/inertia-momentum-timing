% ============================================================
% SLIDE 5: Method A explained
% ============================================================
\begin{frame}{Method A: Markov regime switching}
  \vspace{6pt}
  \emphp{The idea}: assume each factor flips between two unobserved ``moods'': a good one with higher mean returns, and a bad one with lower returns. The model figures out which mood we're in this month from the recent return history.
  \vspace{12pt}
  \begin{center}
  \fbox{\parbox{0.80\textwidth}{\centering
    \textbf{Two states}: ``good'' (high mean) and ``bad'' (low mean)\\[6pt]
    \textbf{Output}: $P(\text{good state today} \mid \text{returns so far})$
  }}
  \end{center}
  \vspace{12pt}
  \begin{itemize}\setlength{\itemsep}{2pt}
    \item Fit by maximum likelihood, one model per FF5 factor
    \item Refit every 12 months on an expanding window (no peeking ahead)
    \item Higher probability $\Rightarrow$ larger weight in that factor next month
  \end{itemize}
\end{frame}

% SLIDE 7: Method B explained
% ============================================================
\begin{frame}{Method B: Ridge predictive regression}
  \vspace{4pt}
  \emphp{The idea}: a simple linear forecast for next month's factor return, using past returns, recent volatility, and macro state (yield curve, credit spread, VIX). Convert the forecast into a probability the factor will outperform.
  \vspace{8pt}
  \begin{center}
  \fbox{\parbox{0.84\textwidth}{\centering
    $\widehat{\text{return}}_{t+1} \;=\; (\text{weights}) \cdot (\text{features}_t)$\\[6pt]
    $P(\text{favorable}) \;=\; \text{sigmoid}\!\left( \widehat{\text{return}}_{t+1} \right)$, \quad rule: hold if $P > 0.5$
  }}
  \end{center}
  \vspace{8pt}
  \begin{itemize}\setlength{\itemsep}{2pt}
    \item Features: own factor lags, cross-factor lags, and macro variables
    \item ``Ridge'' penalty shrinks weights toward zero so the model doesn't overfit
    \item Hold the factor when the model is bullish, otherwise sit in cash
  \end{itemize}
\end{frame}

% SLIDE 9: Method C explained
% ============================================================
\begin{frame}{Method C: gradient-boosted classifier}
  \vspace{4pt}
  \emphp{The idea}: instead of predicting a number, predict a yes/no label (``will this factor beat its median next month?'') using a stack of small decision trees, each correcting the mistakes of the last.
  \vspace{8pt}
  \begin{center}
  \fbox{\parbox{0.84\textwidth}{\centering
    \textbf{Label}: 1 if next-month return $>$ training median, else 0\\[6pt]
    \textbf{Model}: 200 small trees (depth 3) added one by one\\[6pt]
    \textbf{Output}: $P(\text{above median})$, weight $= \max(2P - 1,\, 0)$
  }}
  \end{center}
  \vspace{8pt}
  \begin{itemize}\setlength{\itemsep}{2pt}
    \item Features: cross-factor lags only (macro variables hurt this model)
    \item Captures nonlinear patterns the linear Ridge model can't see
    \item Long-only weight: bigger position when the model is more confident
  \end{itemize}
\end{frame}
